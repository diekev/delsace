set(NOM_PROJET service)

set(INCLUSIONS
	${CMAKE_CURRENT_SOURCE_DIR}
	${CMAKE_CURRENT_SOURCE_DIR}/../
)

add_library(reseau STATIC
	cliente.cc
	cliente_http.cc
	message.cc
	outils_http.cc
	serveuse.cc
	serveuse_http.cc
	uri.cc

	cliente.hh
	cliente_http.hh
	message.hh
	outils_http.hh
	serveuse.hh
	serveuse_http.hh
	uri.hh
)

set(BIBLIOTHEQUES
	base64
	bcrypt
	reseau
	sha256
	ssl
	crypto
)

add_executable(${NOM_PROJET}
    service.cc
)

target_include_directories(${NOM_PROJET} PUBLIC "${INCLUSIONS}")

target_link_libraries(${NOM_PROJET} "${BIBLIOTHEQUES}")

install(
	TARGETS ${NOM_PROJET} RUNTIME DESTINATION .
)

set(NOM_PROJET clientele)

add_executable(${NOM_PROJET}
	clientele.cc
)

target_link_libraries(${NOM_PROJET} "${BIBLIOTHEQUES}")

install(
	TARGETS ${NOM_PROJET} RUNTIME DESTINATION .
)
