set(NOM_BIBLIOTHEQUE math)

set(INCLUSIONS
	${CMAKE_CURRENT_SOURCE_DIR}/../../
)

set(BIBLIOTHEQUES
)

set(ENTETES_MATRICE
	matrice/dimensions.hh
	matrice/matrice.hh
	matrice/matrice_fixe.hh
	matrice/operations.hh
)

set(ENTETES
	aleatoire.hh
	base_vecteur.hh
	boite_englobante.hh
	bruit.hh
	concepts.hh
	entrepolation.hh
	limites.hh
	matrice.hh
