	test_expression.h

	test_structures.hh
	test_types.h
	test_unicode.h
	test_variables.h

	test_decoupage.cc
	test_decoupage_nombres.cc
	test_expression.cc
	test_fonctions.cc
	test_structures.cc
	test_types.cc
	test_unicode.cc
	test_variables.cc

    FILES ${FICHIERS}
	DESTINATION exemples
