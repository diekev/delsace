set(NOM_PROJET test_delsace)

	${TEST_UNITAIRE_INCLUDE_DIRS}
)

	${TEST_UNITAIRE_LIBRARIES}
	reseau
)

	tests.cc
	test_uri.cc

	test_uri.hh
)

	FILES ${FICHIERS}
	DESTINATION exemples
